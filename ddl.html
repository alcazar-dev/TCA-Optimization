lmp_turno_dia NEW Turno diario de limpieza ▼ 
CREATE TABLE lmp_turno_dia (
  fecha              DATE          NOT NULL,        -- PK: fecha de operación
  hora_inicio        TIME          NOT NULL
                                     DEFAULT '07:00',  -- inicio del turno de limpieza
  hora_fin_estimada  TIME          NULL,            -- calculada por el algoritmo
  num_camaristas     SMALLINT      NOT NULL,        -- camaristas activas ese día
  factor_velocidad   DECIMAL(4,2)  NOT NULL
                                     DEFAULT 1.00,    -- ajuste global de velocidad del turno
  cuartos_total      SMALLINT      NOT NULL,        -- cuartos a limpiar ese día
  cuartos_salida     SMALLINT      NOT NULL,        -- con huésped saliente (CHECKOUT)
  cuartos_ocupado    SMALLINT      NOT NULL,        -- sin salida (stayover)
  status             CHAR(1)       NOT NULL
                                     DEFAULT 'A',     -- A=Abierto  C=Cerrado  P=Parcial
  algoritmo_version  VARCHAR(10)   NULL,            -- versión del modelo que generó el scheduling
  usr_cierre         CHAR(3)       NULL,            -- usuario que cerró el turno
  fec_cierre         TIMESTAMP     NULL,

  CONSTRAINT pk_turno_dia  PRIMARY KEY (fecha),
  CONSTRAINT chk_td_status CHECK (status IN ('A','C','P'))
);
lmp_camarista NEW Catálogo de camaristas ▼ 
CREATE TABLE lmp_camarista (
  id_camarista      CHAR(6)       NOT NULL,        -- PK: clave interna de la camarista
  nombre            VARCHAR(60)   NOT NULL,
  zona_habitual     CHAR(3)       NULL,            -- zona/piso asignado normalmente
  min_prom_std      DECIMAL(5,2)  NULL,            -- minutos promedio: habitación estándar
  min_prom_vip      DECIMAL(5,2)  NULL,            -- minutos promedio: habitación VIP
  min_prom_salida   DECIMAL(5,2)  NULL,            -- minutos promedio: limpieza de salida
  obs_totales       INT           NOT NULL
                                    DEFAULT 0,      -- número de observaciones en lmp_tiempo_referencia
  activa            CHAR(1)       NOT NULL
                                    DEFAULT 'S',    -- S/N: disponible para asignación
  usr_alta          CHAR(3)       NOT NULL,
  fec_alta          DATE          NOT NULL,

  CONSTRAINT pk_camarista  PRIMARY KEY (id_camarista),
  CONSTRAINT chk_cam_activa CHECK (activa IN ('S','N'))
);
lmp_asignacion NEW Scheduling diario — tabla central ▼ 
CREATE TABLE lmp_asignacion (
  id_asignacion     BIGINT        NOT NULL GENERATED ALWAYS AS IDENTITY,

  -- Claves foráneas
  fecha             DATE          NOT NULL,        -- FK → lmp_turno_dia.fecha
  num_hab           CHAR(6)       NOT NULL,        -- FK → HOTHAB.c_num_hab  (PMS)
  tpo_hab           CHAR(6)       NULL,            -- FK → HOTCTH.tpo_hab   (desnorm. p/reportes)
  id_camarista      CHAR(6)       NULL,            -- FK → lmp_camarista (NULL = sin asignar)
  res_cve           VARCHAR(15)   NULL,            -- FK → HOTHSP.h_res_cve (próxima llegada)

  -- Algoritmo
  prioridad         SMALLINT      NOT NULL
                                    DEFAULT 50,     -- 1=urgente … 100=baja. Calculado por el algoritmo
  tipo_limpieza     VARCHAR(12)   NOT NULL,       -- SALIDA | OCUPADO | BLOQUEO | INSPECCION
  edif              CHAR(3)       NULL,            -- desnorm. de HOTHAB.c_edif (routing VRP)
  piso              CHAR(3)       NULL,            -- desnorm. de HOTHAB.c_piso (routing VRP)

  -- Tiempos programados
  hora_prog_ini     TIME          NULL,
  hora_prog_fin     TIME          NULL,
  min_estimados     SMALLINT      NULL,            -- del modelo Kaplan-Meier

  -- Tiempos reales (camarista los registra)
  hora_real_ini     TIMESTAMP     NULL,
  hora_real_fin     TIMESTAMP     NULL,
  min_reales        SMALLINT      NULL,            -- calculado al cerrar

  -- Estado y auditoría
  status            VARCHAR(14)   NOT NULL
                                    DEFAULT 'PENDIENTE',
                                    -- PENDIENTE | EN_PROCESO | TERMINADO | INSPECCIONADO | DISPONIBLE | BLOQUEADO
  motivo_reajuste   VARCHAR(20)   NULL,            -- WALKIN | SALIDA_TARD | CAM_BAJA | PRIORIDAD_VIP | MANUAL
  reajuste_cnt      SMALLINT      NOT NULL
                                    DEFAULT 0,      -- veces que el algoritmo reajustó este bloque
  usr_mod           CHAR(3)       NULL,
  fec_mod           TIMESTAMP     NULL,

  CONSTRAINT pk_asignacion       PRIMARY KEY (id_asignacion),
  CONSTRAINT uq_fecha_hab        UNIQUE (fecha, num_hab),   -- un cuarto, una asignación por día
  CONSTRAINT fk_asi_turno        FOREIGN KEY (fecha)
                                     REFERENCES lmp_turno_dia (fecha),
  CONSTRAINT fk_asi_camarista    FOREIGN KEY (id_camarista)
                                     REFERENCES lmp_camarista (id_camarista),
  CONSTRAINT chk_asi_tipo        CHECK (tipo_limpieza IN ('SALIDA','OCUPADO','BLOQUEO','INSPECCION')),
  CONSTRAINT chk_asi_status      CHECK (status IN ('PENDIENTE','EN_PROCESO','TERMINADO',
                                               'INSPECCIONADO','DISPONIBLE','BLOQUEADO'))
);

CREATE INDEX idx_asi_fecha_cam    ON lmp_asignacion (fecha, id_camarista);  -- horario por camarista
CREATE INDEX idx_asi_fecha_piso   ON lmp_asignacion (fecha, edif, piso); -- routing por piso
CREATE INDEX idx_asi_status       ON lmp_asignacion (fecha, status);       -- semáforo en tiempo real
lmp_tiempo_referencia NEW Historial para el modelo predictivo ▼ 
CREATE TABLE lmp_tiempo_referencia (
  id_obs            BIGINT        NOT NULL GENERATED ALWAYS AS IDENTITY,

  -- Contexto de la observación
  id_asignacion     BIGINT        NOT NULL,        -- FK → lmp_asignacion
  fecha             DATE          NOT NULL,        -- desnorm. para queries analíticos
  num_hab           CHAR(6)       NOT NULL,
  tpo_hab           CHAR(6)       NOT NULL,        -- FK → HOTCTH.tpo_hab
  id_camarista      CHAR(6)       NOT NULL,        -- FK → lmp_camarista
  tipo_limpieza     VARCHAR(12)   NOT NULL,

  -- Covariables del modelo Kaplan-Meier / XGBoost
  min_reales        SMALLINT      NOT NULL,        -- variable objetivo
  num_personas_prev SMALLINT      NOT NULL
                                    DEFAULT 0,      -- personas en la estancia que salió
  cama_extra        CHAR(1)       NOT NULL
                                    DEFAULT 'N',
  cuna              CHAR(1)       NOT NULL
                                    DEFAULT 'N',
  num_reqs_especiales SMALLINT    NOT NULL
                                    DEFAULT 0,      -- count de HOTREQ pendientes
  tipo_salida       CHAR(1)       NULL,            -- W=walk-in previo  N=normal  A=anticipada
  dia_semana        SMALLINT      NOT NULL,        -- 1=lun … 7=dom (feature temporal)
  noches_estancia   SMALLINT      NULL,            -- más noches = más desorden
  fac_ama_tipo      SMALLINT      NULL,            -- desnorm. de HOTCTH.fac_ama

  CONSTRAINT pk_tref             PRIMARY KEY (id_obs),
  CONSTRAINT fk_tref_asignacion  FOREIGN KEY (id_asignacion)
                                     REFERENCES lmp_asignacion (id_asignacion),
  CONSTRAINT fk_tref_camarista   FOREIGN KEY (id_camarista)
                                     REFERENCES lmp_camarista (id_camarista),
  CONSTRAINT chk_tref_min        CHECK (min_reales BETWEEN 1 AND 480)
);

CREATE INDEX idx_tref_camarista  ON lmp_tiempo_referencia (id_camarista, tpo_hab);
CREATE INDEX idx_tref_tpohab     ON lmp_tiempo_referencia (tpo_hab, fecha);
lmp_prediccion_demanda NEW Forecast de ocupación (Parte 2) ▼ 
CREATE TABLE lmp_prediccion_demanda (
  id_pred           BIGINT        NOT NULL GENERATED ALWAYS AS IDENTITY,

  fecha_pred        DATE          NOT NULL,        -- fecha para la que se predice
  tpo_hab           CHAR(6)       NULL,            -- FK → HOTCTH.tpo_hab  (NULL = total hotel)

  -- Salida del modelo (SARIMA / Prophet / XGBoost / LightGBM)
  ocu_estimada      DECIMAL(5,2)  NOT NULL,        -- cuartos ocupados estimados
  ocu_pct           DECIMAL(5,2)  NULL,            -- porcentaje de ocupación estimada
  lld_estimadas     DECIMAL(5,2)  NOT NULL,        -- llegadas previstas
  sda_estimadas     DECIMAL(5,2)  NOT NULL,        -- salidas previstas
  carga_limpieza    DECIMAL(6,2)  NULL,            -- minutos totales estimados de limpieza
  ic_inferior       DECIMAL(5,2)  NULL,            -- intervalo de confianza inferior (95%)
  ic_superior       DECIMAL(5,2)  NULL,            -- intervalo de confianza superior (95%)

  -- Metadata del modelo
  modelo_usado      VARCHAR(20)   NOT NULL,        -- SARIMA | PROPHET | XGBOOST | LGBM | ENSEMBLE
  horizonte_dias    SMALLINT      NOT NULL,        -- días hacia adelante desde la fecha de generación
  generado_en       TIMESTAMP     NOT NULL
                                    DEFAULT CURRENT_TIMESTAMP,
  mae               DECIMAL(5,3)  NULL,            -- error del modelo en backtesting
  rmse              DECIMAL(5,3)  NULL,

  CONSTRAINT pk_pred        PRIMARY KEY (id_pred),
  CONSTRAINT uq_pred_fecha  UNIQUE (fecha_pred, tpo_hab, modelo_usado, generado_en)
);

CREATE INDEX idx_pred_fecha  ON lmp_prediccion_demanda (fecha_pred, tpo_hab);
HOTHSP PMS Columnas usadas por el módulo ▼ 
-- Referencia PMS — solo lectura. Columnas consumidas por el módulo LMP.
SELECT
  h_res_cve,       -- clave de reservación (join con lmp_asignacion.res_cve)
  h_num_hab,       -- número de habitación
  h_tpo_hab,       -- tipo de habitación
  h_status,        -- 00=reserv 10=en casa 50=salida 02=no-show
  h_fec_lld,       -- fecha llegada  → ventana de tiempo disponible
  h_hra_lld,       -- hora llegada   → deadline de limpieza
  h_fec_sda,       -- fecha salida   → inicio ventana de limpieza
  h_hra_sda,       -- hora salida    → inicio real disponible
  h_num_per,       -- personas       → covariable modelo tiempos
  h_num_adu,
  h_num_men,
  h_tpo_plan,      -- plan (AI/AP/MAP…) → afecta intensidad de uso
  h_ult_cam_fec    -- último cambio  → detectar llegadas walk-in tardías
FROM HOTHSP
WHERE h_fec_lld = CURRENT_DATE
  OR h_fec_sda  = CURRENT_DATE
  OR (h_status = '10' AND h_fec_sda > CURRENT_DATE);
-- (stayovers que siguen en casa y necesitan limpieza de ocupado)

  c_tpo_hab,        -- tipo → join con HOTCTH

HOTCTH PMS Columnas usadas por el módulo ▼ 
-- Referencia PMS — solo lectura.
SELECT
  tpo_hab,    -- PK del tipo
  descrip,    -- etiqueta visual en el horario
  fac_ama,    -- proxy de complejidad de limpieza (tiempo base)
  cupo,       -- 1=individual 2=doble… → afecta desgaste
  clase,      -- clasificación → define nivel de prioridad VIP
  hab_dis     -- inventario total del tipo
FROM HOTCTH;
HOTREQ PMS Columnas usadas por el módulo ▼ 
-- Referencia PMS — solo lectura. Requerimientos del día para el bloque visual.
SELECT
  res_cve,          -- join con lmp_asignacion.res_cve
  cve_req,           -- código del requerimiento
  desc_req,          -- texto visible en el bloque horario
  tipo,              -- R=Requerimiento B=Beneficio E=ENZO
  fecha_ejec,        -- filtro: solo los del día
  pendiente,         -- S=por hacer → ícono activo en horario
  num_art,           -- cantidad de artículos → tiempo adicional
  depto              -- departamento responsable
FROM HOTREQ
WHERE fecha_ejec = CURRENT_DATE
  AND pendiente   = 'S';
